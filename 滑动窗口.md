对于下面一种1208 的题解，或许能够对 **问题2** 进行一波解释，并没有维护res

```c++
class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int left=0, right=0, cost=0;
        while (right < s.size()) {
            cost += abs(s[right] - t[right++]);
            // 不满足条件时左指针只右移一次，保持窗口不缩小
            if (cost > maxCost) cost -= abs(s[left] - t[left++]);
        }
        return right - left;
    }
};
```
